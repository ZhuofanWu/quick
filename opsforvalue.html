Redis Value 操作示例

set(K key, V value)

设置键值对

redisTemplate.opsForValue().set("key", "value");

setEx(K key, V value, long timeout, TimeUnit unit)

设置键值对并指定过期时间

redisTemplate.opsForValue().set("key", "value", 60, TimeUnit.SECONDS);

get(K key)

获取键对应的值

String value = (String) redisTemplate.opsForValue().get("key");

increment(K key, long delta)

数值递增

Long newValue = redisTemplate.opsForValue().increment("counter", 1);

decrement(K key, long delta)

数值递减

Long newValue = redisTemplate.opsForValue().decrement("counter", 1);

delete(K key)

删除键值对

Boolean deleted = redisTemplate.delete("key");

multiSet(Map<? extends K, ? extends V> map)

批量设置多个键值对

Map<String, String> map = new HashMap<>();
map.put("key1", "value1");
map.put("key2", "value2");
redisTemplate.opsForValue().multiSet(map);

multiGet(Collection<K> keys)

批量获取多个键的值

List<String> keys = Arrays.asList("key1", "key2");
List<Object> values = redisTemplate.opsForValue().multiGet(keys);
